async function validateAndInsert(
  project: Record<string, unknown>,
  existingNames: Set<string>,
): Promise<{ inserted: boolean; reason?: string }> {
  const name = String(project.projectName ?? "").trim();
  const rawCountry = String(project.country ?? "").trim();
  const description = String(project.description ?? "");

  // Basic field validation
  if (!name || name.length < 5) return { inserted: false, reason: "Name too short" };
  if (existingNames.has(name.toLowerCase())) return { inserted: false, reason: "Duplicate" };

  // Country validation
  const country = normalizeCountry(rawCountry);
  if (!country) return { inserted: false, reason: `Non-African country: ${rawCountry}` };

  // Non-energy exclusion
  if (isLikelyNonEnergy(name, description)) return { inserted: false, reason: `Non-energy project: ${name}` };

  // Deal size sanity
  const dealSize = sanitizeDealSize(project.dealSizeUsdMn);

  // Technology validation
  const tech = String(project.technology ?? "Other Renewables");
  const validTech = VALID_TECHNOLOGIES.has(tech) ? tech : "Other Renewables";

  try {
    await db.insert(projectsTable).values({
      projectName: name,
      country,
      region: String(project.region ?? inferRegion(country)),
      technology: validTech,
      dealSizeUsdMn: dealSize,
      investors: typeof project.investors === "string" ? project.investors : null,
      status: String(project.status ?? "announced"),
      description: description || null,
      capacityMw: typeof project.capacityMw === "number" ? project.capacityMw : null,
      announcedYear: typeof project.announcedYear === "number" ? project.announcedYear : new Date().getFullYear(),
      closedYear: null,

      sourceUrl: typeof project.sourceUrl === "string" ? project.sourceUrl : null,
      newsUrl: typeof project.newsUrl === "string" ? project.newsUrl : null,
      isAutoDiscovered: true,
      reviewStatus: "pending",
      discoveredAt: new Date(),
    });
    existingNames.add(name.toLowerCase());
    return { inserted: true };
  } catch (err) {
    return { inserted: false, reason: `DB error: ${err instanceof Error ? err.message : String(err)}` };
  }
}
